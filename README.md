### 빌드 전 mysql 설치 및 설정(필수) 
#### mac     
        brew install mysql  
        mysql -u root -p
        create database miniproj1; 
        create user 'admin'@'localhost' identified by 'qwer1234';  
        grant all privileges on miniproj1.* to 'admin'@'qwer1234'; 
        flush privleges;   



#### ubuntu 
            sudo apt install mysql-server   
            sudo mysql -u root -p 
            create database miniproj1;  
            create user 'admin'@'localhost' identified by 'qwer1234'; 
            grant all privileges on miniproj1.* to 'admin'@'qwer1234';
            flush privleges;
